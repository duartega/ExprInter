
def findSmallestIdxFromIdx(names, idx):
    smallestIdx = idx
    for i in range(idx + 1, len(names)):
        if names[i] < names[idx]:
            smallestIdx = i
    return smallestIdx

        
def insertionSort(names):
    upperBound = len(names) - 1
    for idx in range( upperBound ):
        smallestIdx = findSmallestIdxFromIdx(names, idx)
        if smallestIdx != idx:
            tmp = names[idx]
            names[idx] = names[smallestIdx]
            names[smallestIdx] = tmp
